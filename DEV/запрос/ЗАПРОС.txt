USE DEV
GO
WITH C AS
(
SELECT
     dayId
    ,artId
    ,cntrId
    ,endQnty
	,DATEDIFF(d, dayId, DENSE_RANK() OVER(PARTITION BY artId,cntrId ORDER BY dayId)) AS diff
FROM dbo.tTestTable1
)
SELECT 
     dayId
    ,artId
    ,cntrId
    ,endQnty
    ,ROW_NUMBER() OVER (PARTITION BY artId,cntrId,diff ORDER BY dayId)
FROM C;
GO
