WITH table1 AS
(
SELECT
     normalDt
    ,skuid
    ,dcid
	,tryeDayPcs
	,DATEDIFF(d, DENSE_RANK() OVER(
                                   PARTITION BY
								        skuid
									   ,dcid
                                   ORDER BY
								       normalDt), normalDt) AS diff
FROM #tmpLoadPost
)
SELECT 
     normalDt
    ,skuId
    ,dcId
	,tryeDayPcs
    ,ROW_NUMBER() OVER (
	                    PARTITION BY
						     skuid
							,dcid
							,diff
						ORDER BY
						    normalDt) AS withoutShipment
FROM table1;
